% Redefine emph to start with bold instead of italic. Can still be nested.
% See http://tex.stackexchange.com/a/6757/100641.
\makeatletter
\DeclareRobustCommand{\em}{%
	\@nomath\em \if b\expandafter\@car\f@series\@nil
	\normalfont \else \bfseries \fi}
\makeatother
